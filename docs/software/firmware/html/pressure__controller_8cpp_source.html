275 void PressureController::computePlateau(uint16_t p_centiSec) {
276  uint16_t minValue = m_lastPressureValues[0u];
277  uint16_t maxValue = m_lastPressureValues[0u];
278  uint16_t totalValues = m_lastPressureValues[0u];
279 
280  for (uint8_t index = 1u; index < MAX_PRESSURE_SAMPLES; index++) {
281  minValue = min(minValue, m_lastPressureValues[index]);
282  maxValue = max(maxValue, m_lastPressureValues[index]);
283  totalValues += m_lastPressureValues[index];
284  }
285 
286  uint16_t diff = (maxValue - minValue);
287 
288  // Start computing plateau pressure when:
289  // - the last pressure values were close enough
290  // - the hold inspiration phase is about to end
291  // - plateau pressure computation was not already started
292  if (!m_plateauComputed && (diff < 10u)
293  && (p_centiSec >= ((m_centiSecPerInhalation * 95u) / 100u))) {
294  m_startPlateauComputation = true;
295  }
296 
297  // Stop computing plateau pressure when pressure drops
298  if (m_startPlateauComputation && (diff > 10u)) {
299  m_startPlateauComputation = false;
300  m_plateauComputed = true;
301  }
302 
303  if (m_startPlateauComputation) {
304  m_plateauPressure = totalValues / MAX_PRESSURE_SAMPLES;
305  }
306 }

454 void PressureController::updatePeakPressure() {
455  int16_t plateauDelta = m_maxPlateauPressureCommand - m_plateauPressure;
456  int16_t peakDelta = m_maxPeakPressureCommand - m_peakPressure;
457 
458  DBG_DO(Serial.println("updatePeakPressure");)
459 
460  // If a plateau was detected
461  if ((m_plateauPressure > 0u) && (m_plateauPressure < UINT16_MAX)) {
462  DBG_DO(Serial.println("Plateau detected");)
463 
464  if (abs(plateauDelta) > 20) {
465  m_maxPeakPressureCommand =
466  // cppcheck-suppress misra-c2012-12.3
467  min((min(m_peakPressure, m_maxPeakPressureCommand) + plateauDelta),
468  static_cast<int>(CONST_MAX_PEAK_PRESSURE));
469  } else if ((abs(plateauDelta) < 20) && (abs(plateauDelta) > 5)) {
470  m_maxPeakPressureCommand =
471  min(min(m_peakPressure, m_maxPeakPressureCommand) + (plateauDelta / 2),
472  static_cast<int>(CONST_MAX_PEAK_PRESSURE));

481  DBG_DO(Serial.println(m_plateauStartTime);)
482 
483  // If plateau was reached quite early

485  DBG_DO(Serial.println("BLOWER -20");)

503  DBG_DO(Serial.println("BLOWER +10");)

558 void PressureController::checkCycleAlarm() {
559  // RCM-SW-1 + RCM-SW-14 : Check if plateau is reached
560  uint16_t minPlateauBeforeAlarm =
561  (m_maxPlateauPressureCommand * (100u - ALARM_THRESHOLD_DIFFERENCE_PERCENT)) / 100u;
562  uint16_t maxPlateauBeforeAlarm =
563  (m_maxPlateauPressureCommand * (100u + ALARM_THRESHOLD_DIFFERENCE_PERCENT)) / 100u;
564  if ((m_plateauPressure < minPlateauBeforeAlarm)
565  || (m_plateauPressure > maxPlateauBeforeAlarm)) {
566  m_alarmController->detectedAlarm(RCM_SW_1, m_cycleNb, m_maxPlateauPressureCommand,
567  m_pressure);
568  m_alarmController->detectedAlarm(RCM_SW_14, m_cycleNb, m_maxPlateauPressureCommand,
569  m_pressure);
570  } else {
571  m_alarmController->notDetectedAlarm(RCM_SW_1);
572  m_alarmController->notDetectedAlarm(RCM_SW_14);
573  }
574 
575  // RCM-SW-2 + RCM-SW-19 : Check is mean pressure was < 2 cmH2O
576  uint16_t meanPressure = m_sumOfPressures / m_numberOfPressures;
577  if (meanPressure <= ALARM_THRESHOLD_MIN_PRESSURE) {
578  m_alarmController->detectedAlarm(RCM_SW_2, m_cycleNb, ALARM_THRESHOLD_MIN_PRESSURE,
579  m_pressure);

607 int32_t PressureController::pidBlower(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
608  // Compute error
609  int32_t error = targetPressure - currentPressure;
610 
611  // Compute integral
612  blowerIntegral = blowerIntegral + ((PID_BLOWER_KI * error * dt) / 1000000);
613  blowerIntegral = max(PID_BLOWER_INTEGRAL_MIN, min(PID_BLOWER_INTEGRAL_MAX, blowerIntegral));
614 
615  // Compute derivative
616  int32_t derivative = ((blowerLastError == INVALID_ERROR_MARKER) || (dt == 0))
617  ? 0
618  : ((1000000 * (error - blowerLastError)) / dt);
619  blowerLastError = error;
620 
621  int32_t blowerCommand = (PID_BLOWER_KP * error) + blowerIntegral
622  + ((PID_BLOWER_KD * derivative) / 1000); // Command computation
623 
624  int32_t minAperture = m_blower_valve.minAperture();
625  int32_t maxAperture = m_blower_valve.maxAperture();
626 
627  uint32_t blowerAperture =
628  max(minAperture,
629  min(maxAperture, maxAperture + (minAperture - maxAperture) * blowerCommand / 1000));
630 
631  return blowerAperture;
632 }

634 int32_t
635 PressureController::pidPatient(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
636  // Compute error
637  int32_t error = targetPressure + PID_PATIENT_SAFETY_PEEP_OFFSET - currentPressure;
638 
639  // Compute integral
640  patientIntegral = patientIntegral + ((PID_PATIENT_KI * error * dt) / 1000000);
641  patientIntegral = max(PID_PATIENT_INTEGRAL_MIN, min(PID_PATIENT_INTEGRAL_MAX, patientIntegral));
642 
643  // Compute derivative
644  int32_t derivative = ((patientLastError == INVALID_ERROR_MARKER) || (dt == 0))
645  ? 0
646  : ((1000000 * (error - patientLastError)) / dt);
647  patientLastError = error;
648 
649  int32_t patientCommand = (PID_PATIENT_KP * error) + patientIntegral
650  + ((PID_PATIENT_KD * derivative) / 1000); // Command computation
651 
652  int32_t minAperture = m_blower_valve.minAperture();
653  int32_t maxAperture = m_blower_valve.maxAperture();
654 
655  uint32_t patientAperture =
656  max(minAperture,
657  min(maxAperture, maxAperture + (maxAperture - minAperture) * patientCommand / 1000));
658 
659  return patientAperture;
